import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from scipy.stats import pearsonr
from scipy.stats import pointbiserialr

# Function to determine correlation strength
def get_correlation_strength(corr):
    if abs(corr) > 0.8:
        return "Strong correlation"
    elif 0.5 <= abs(corr) <= 0.8:
        return "Moderate correlation"
    elif 0.3 <= abs(corr) < 0.5:
        return "Weak correlation"
    else:
        return "Very weak or no correlation"

# function for non binary synergy
def corr_non_binary(df, metrics):
    # Initialize a dictionary to store correlation results
    correlation_results = {}

    # Create scatter plots to visualize each metric against synergy
    for metric in metrics:
        plt.figure(figsize=(6, 4))
        sns.scatterplot(data=df, x=metric, y="synergy")
        plt.axhline(0, color='red', linestyle='--', linewidth=1, label='Synergy = 0')  # Add red line
        plt.title(f"Synergy vs {metric}")
        plt.xlabel(metric)
        plt.ylabel("Synergy")
        plt.legend()
        plt.show()
        
        # Calculate Pearson correlation coefficient
        corr, p_value = pearsonr(df[metric], df["synergy"])
        correlation_results[metric] = {"correlation": corr, "p-value": p_value}



    # Summarize and print the results
    print("\nCorrelation Analysis Summary:")
    for metric, results in correlation_results.items():
        correlation = results["correlation"]
        p_value = results["p-value"]
        significance = "Significant" if p_value < 0.05 else "Not Significant"
        strength = get_correlation_strength(correlation)
        print(
            f"\033[1m{metric}:\033[0m\n"
            f"  Correlation Coefficient: {correlation:.2f}\n"
            f"  P-Value: {p_value:.2e}\n"
            f"  Strength of Correlation: \033[1m{strength}\033[0m\n"
            f"  Significance: \033[1m{significance}\033[0m"
        )

# function for binary synergy
def corr_binary(df, metrics):
    # Initialize a dictionary to store correlation results
    correlation_results = {}

    # Create scatter plots to visualize each metric against the binary synergy
    for metric in metrics:
        
        # Calculate Point-Biserial correlation coefficient
        corr, p_value = pointbiserialr(df["synergistic"], df[metric])
        correlation_results[metric] = {"correlation": corr, "p-value": p_value}

    # Function to determine correlation strength
    def get_correlation_strength(corr):
        if abs(corr) > 0.8:
            return "Strong correlation"
        elif 0.5 <= abs(corr) <= 0.8:
            return "Moderate correlation"
        elif 0.3 <= abs(corr) < 0.5:
            return "Weak correlation"
        else:
            return "Very weak or no correlation"

    # Summarize and print the results
    print("\nCorrelation Analysis Summary (Binary Synergy):")
    for metric, results in correlation_results.items():
        correlation = results["correlation"]
        p_value = results["p-value"]
        significance = "Significant" if p_value < 0.05 else "Not Significant"
        strength = get_correlation_strength(correlation)  
        print(
            f"\033[1m{metric}:\033[0m\n"
            f"  Correlation Coefficient: {correlation:.2f}\n"
            f"  P-Value: {p_value:.2e}\n"
            f"  Strength of Correlation: \033[1m{strength}\033[0m\n"
            f"  Significance: \033[1m{significance}\033[0m"
        )
